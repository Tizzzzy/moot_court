================================================================================
                          API SPECIFICATION
================================================================================

POST /api/ocr/upload
  Description: Upload PDF for OCR processing
  Parameters:
    - file (multipart/form-data): PDF file
    - user_id (form field): User identifier
  Returns: {job_id: string, status: string, message: string}
  Status Codes: 200 (success), 400 (invalid file), 413 (too large)
  Background Task: Runs MinerU → Qwen extraction pipeline

GET /api/ocr/status/{job_id}
  Description: Poll job completion status
  Parameters: job_id (path)
  Returns: {job_id, status, created_at, case_id?, completed_at?, error?}
  Status Codes: 200 (success), 404 (not found)
  Statuses: pending | processing | completed | failed

GET /api/cases/{case_id}
  Description: Retrieve case details with parties
  Parameters: case_id (path)
  Returns: {id, case_number, case_type, state, plaintiffs[], defendants[], ...}
  Status Codes: 200 (success), 404 (not found)
  Response Includes: All case details + linked party information

GET /api/health
  Description: Health check endpoint
  Returns: {status: "healthy"}
  Status Codes: 200

================================================================================
                        DATABASE SCHEMA
================================================================================

TABLE: processing_jobs
  Columns:
    - id (INTEGER PRIMARY KEY)
    - job_id (VARCHAR(36) UNIQUE NOT NULL)
    - job_type (ENUM: ocr, evidence_recommend, evidence_validate)
    - status (ENUM: pending, processing, completed, failed)
    - case_id (INTEGER FK nullable)
    - input_file_path (TEXT)
    - output_data (TEXT - JSON)
    - error_message (TEXT)
    - created_at (DATETIME)
    - completed_at (DATETIME)
  Indexes: job_id

TABLE: cases
  Columns:
    - id (INTEGER PRIMARY KEY)
    - user_id (VARCHAR(50))
    - case_number (VARCHAR(100))
    - case_type (VARCHAR(100))
    - state (VARCHAR(50))
    - filing_date (DATE)
    - claim_summary (TEXT)
    - amount_sought (DECIMAL)
    - incident_date (DATE)
    - status (VARCHAR(20))
    - created_at (DATETIME)
  Indexes: user_id
  Relations: parties

TABLE: parties
  Columns:
    - id (INTEGER PRIMARY KEY)
    - case_id (INTEGER FK)
    - role (VARCHAR(20): plaintiff|defendant)
    - name (VARCHAR(255))
    - address (TEXT)
  Indexes: case_id
